    function isValidPieces(faceletArray) {
        // 1. Định nghĩa các cặp màu đối nhau không bao giờ chung 1 viên
        const opposites = { 'U':'D', 'D':'U', 'F':'B', 'B':'F', 'R':'L', 'L':'R' };
        
        // 2. Định nghĩa Index của 12 cặp Cạnh trong mảng 54 facelet
        // (Lấy 2 vị trí liền kề ghép thành 1 cạnh. Ví dụ UR, UF, UL, UB...)
        const edges = [
            [7, 25], [5, 16], [1, 46], [3, 37], // Các cạnh mặt U
            [28, 19], [32, 14], [34, 43], [30, 52], // Các cạnh mặt D
            [21, 12], [23, 41], [48, 9], [50, 39] // Các cạnh giữa (Equator)
        ];
    
        // 3. Định nghĩa Index của 8 Góc (mỗi góc 3 mặt)
        const corners = [
            [8, 26, 15], [2, 17, 45], [0, 47, 36], [6, 38, 24], // Góc trên
            [27, 20, 11], [29, 18, 44], [35, 42, 53], [33, 51, 14] // Góc dưới
        ];
    
        // Kiểm tra Cạnh
        for (let [i1, i2] of edges) {
            let c1 = faceletArray[i1], c2 = faceletArray[i2];
            if (c1 === c2) return false; // Cạnh có 2 màu giống nhau
            if (opposites[c1] === c2) return false; // Cạnh chứa 2 màu đối nhau (VD: Đỏ - Cam)
        }
    
        // Kiểm tra Góc
        for (let [i1, i2, i3] of corners) {
            let c1 = faceletArray[i1], c2 = faceletArray[i2], c3 = faceletArray[i3];
            let colors = [c1, c2, c3];
            // Góc có màu trùng nhau
            if (new Set(colors).size < 3) return false; 
            // Góc chứa màu đối nhau
            if (opposites[c1] === c2 || opposites[c1] === c3 || opposites[c2] === c3) return false; 
        }
    
        return true; // Tất cả các viên đều có màu hợp lệ!
    }

        def verify(self):
            # 1. Kiểm tra thiếu hoặc trùng viên
            edge_pieces = set(e['e'] for e in self.edges)
            if len(edge_pieces) != 12 or -1 in edge_pieces:
                raise ValueError("Lỗi cấu trúc Cạnh: Thiếu viên, trùng viên, hoặc sai màu.")
                
            corner_pieces = set(c['c'] for c in self.corners)
            if len(corner_pieces) != 8 or -1 in corner_pieces:
                raise ValueError("Lỗi cấu trúc Góc: Thiếu viên, trùng viên, hoặc sai màu.")
    
            # 2. Kiểm tra lật cạnh (Flip parity) -> Tổng lật phải chẵn
            edge_o_sum = sum(e['o'] for e in self.edges)
            if edge_o_sum % 2 != 0:
                raise ValueError("Lỗi lật Cạnh (Edge flip): Có viên cạnh bị lật ngược.")
    
            # 3. Kiểm tra lật góc (Twist parity) -> Tổng xoắn phải chia hết cho 3
            corner_o_sum = sum(c['o'] for c in self.corners)
            if corner_o_sum % 3 != 0:
                raise ValueError("Lỗi lật Góc (Corner twist): Có viên góc bị xoắn sai chiều.")
    
            # 4. Kiểm tra nghịch thế chẵn lẻ (Permutation Parity) -> Hoán vị góc và cạnh phải khớp
            edge_inv = sum(1 for i in range(11) for j in range(i + 1, 12) if self.edges[i]['e'] > self.edges[j]['e'])
            corner_inv = sum(1 for i in range(7) for j in range(i + 1, 8) if self.corners[i]['c'] > self.corners[j]['c'])
    
            if (edge_inv % 2) != (corner_inv % 2):
                raise ValueError("Lỗi hoán vị (Parity): Không thể giải do có 2 viên góc hoặc cạnh bị hoán đổi trái phép.")
                
            return True

  Bạn hoàn toàn yên tâm, backend  web_api.py  hiện tại ĐÃ đảm bảo việc xử lý nhiều request cùng lúc rất tốt rồi, bạn không cần thiết phải tốn công viết thêm Queue phức tạp nữa đâu!

  Dưới đây là cách mà cấu trúc hiện tại đang xử lý đa luồng/đa tiến trình cực kỳ hiệu quả:

  ### 1. Khả năng chịu tải (Concurrency) hiện tại:

  • FastAPI & Uvicorn: Bản thân FastAPI là một framework bất đồng bộ (Asynchronous). Nó có thể nhận hàng ngàn request cùng một lúc mà không bị block (nghẽn).
  • Cơ chế  multiprocessing.Process : Mỗi khi có một người dùng bấm nút "Giải ngay", backend của bạn sẽ tạo ra một Process con (Tiến trình riêng biệt) hoàn toàn mới để tính toán. Điều
  này giúp tận dụng tối đa tất cả các nhân (Cores) của CPU. Nếu 4 người cùng bấm giải, 4 nhân CPU sẽ cùng lúc tính toán độc lập.
  • Tối ưu RAM bằng  fork  (Copy-on-Write): Vì các bảng Lookup Table (chiếm nhiều RAM) đã được load vào biến global ngay khi bật server, nên khi Linux sinh ra Process con (bằng cơ chế
  fork ), các Process con này sẽ dùng chung vùng nhớ chứa Table với Process cha mà không phải tốn RAM để load lại từ đầu.

  ### 2. Vậy khi nào mới thực sự CẦN dùng tới Queue (Hàng đợi)?

  Cấu trúc hiện tại (sinh Process mới cho mỗi request) là vô cùng hoàn hảo cho dự án cá nhân hoặc web có lượng truy cập vừa và nhỏ.

  Bạn CHỈ cần đưa Queue (như Celery, Redis Queue, RabbitMQ) hoặc dùng  ProcessPoolExecutor  vào khi:

  • Web của bạn bất ngờ nổi tiếng, có hàng nghìn người dùng bấm nút Giải cùng một phần nghìn giây. Việc sinh ra 1000 Process con cùng lúc sẽ vắt kiệt CPU và làm sập máy chủ.
  • Lúc đó, Queue sẽ có tác dụng: "Máy chủ tôi chỉ có 4 nhân CPU, nên tôi chỉ giải cho 4 người đầu tiên. 996 người còn lại vui lòng nằm trong Hàng Đợi (Queue), ai xong thì người tiếp
  theo mới được vào".

  Tóm lại:
  Với nhu cầu hiện tại, logic xử lý của bạn đã là kiến trúc rất tiêu chuẩn và xuất sắc (vừa có hard timeout 10 giây để chống treo máy, vừa có multiprocessing để chạy song song, vừa
  tận dụng CoW để tiết kiệm RAM). Bạn không cần phải code thêm Queue rườm rà nữa nhé!
